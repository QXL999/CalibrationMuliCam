ThreeFrequencyUnwrapping::ThreeFrequencyUnwrapping(int width , int height,std::vector<int> freq,std::string imgpath)
{
    width_ = (width);
    height_ = (height);
    freq_ = (freq);
	imgpath_ = (imgpath);
    //freq_ = { 81, 72, 64 };
}
// 生成正弦图案
cv::Mat ThreeFrequencyUnwrapping::generateSinePattern(int freq_index, int phase_shift,bool index)
{
    cv::Mat pattern(height_, width_, CV_32FC1);
    if (!index)
    {
        float phase = phase_shift * CV_PI / 2.0f;//初始相位偏移
        for (int k = 0; k < width_; k++)
        {
            float value = 128.0f + 127.0f * sin(2 * CV_PI * k * freq_[freq_index] / width_ + phase);
            for (int g = 0; g < height_; g++)
            {
                pattern.at<float>(g, k) = value;
            }
        }
    }
    else
    {
        // 实际应用中，这里应该替换为从相机采集的图像
        // 例如：pattern = cv::imread("captured_image.png", cv::IMREAD_GRAYSCALE);
		//std::string filename = imgpath_ + std::to_string(freq_[freq_index]) + "_" + std::to_string(phase_shift) + ".png";
        std::string filename = imgpath_ + std::to_string(freq_index) + "_" + std::to_string(phase_shift) + ".bmp";
        pattern = cv::imread(filename, cv::IMREAD_GRAYSCALE);
    }
    
    return pattern;
}
// 对图像进行归一化处理 [0, 1] 范围
cv::Mat ThreeFrequencyUnwrapping::normalizeImage(const cv::Mat& img)
{
    cv::Mat normalized;
    double minVal, maxVal;
    cv::minMaxLoc(img, &minVal, &maxVal);

    if (maxVal > minVal) 
    {
        img.convertTo(normalized, CV_32F, 1.0 / (maxVal - minVal), -minVal / (maxVal - minVal));
    }
    else 
    {
        normalized = cv::Mat::ones(img.size(), CV_32F);
    }

    return normalized;
}
cv::Mat ThreeFrequencyUnwrapping::computeWrappedPhase(const cv::Mat& I1, const cv::Mat& I2, const cv::Mat& I3, const cv::Mat& I4)
{
    cv::Mat phase = cv::Mat::zeros(height_, width_, CV_32F);


    //// 使用四步法计算相位
    //for (int y = 0; y < height_; y++)
    //{
    //    for (int x = 0; x < width_;x++)
    //    {
    //        float I11 = I1.at<float>(y, x);
    //        float I22 = I2.at<float>(y, x);
    //        float I33 = I3.at<float>(y, x);
    //        float I44 = I4.at<float>(y, x);
    //        // 计算四步法的相位
    //        float wrappedPhase = atan2(I44 - I22, I11 - I33); // Calculate phase using the four-step method
    //        //wrappedPhase.at<double>(y, x) = phase;
    //        if (wrappedPhase < 0)
    //            phase.at<float>(y, x) = wrappedPhase + 2 * CV_PI;
    //        else
    //            phase.at<float>(y, x) = wrappedPhase;
    //    }
    //}

    for (int g = 0; g < height_; g++) 
    {
        for (int k = 0; k < width_; k++) 
        {
            float i1 = I1.at<float>(g, k);
            float i2 = I2.at<float>(g, k);
            float i3 = I3.at<float>(g, k);
            float i4 = I4.at<float>(g, k);

            // 处理四个特殊位置,针对正弦条纹
            if (std::abs(i4 - i2) < 1e-6 && i1 > i3) 
            {
                phase.at<float>(g, k) = 0;
            }
            else if (std::abs(i4 - i2) < 1e-6 && i1 < i3) 
            {
                phase.at<float>(g, k) = CV_PI;
            }
            else if (std::abs(i1 - i3) < 1e-6 && i4 > i2) 
            {
                phase.at<float>(g, k) = CV_PI / 2;
            }
            else if (std::abs(i1 - i3) < 1e-6 && i4 < i2) 
            {
                phase.at<float>(g, k) = 3 * CV_PI / 2;
            }
            // 处理各个象限
            else if (i1 < i3) { // 第二、三象限
                phase.at<float>(g, k) = std::atan((i4 - i2) / (i1 - i3)) + CV_PI;
            }
            else if (i1 > i3 && i4 > i2) { // 第一象限
                phase.at<float>(g, k) = std::atan((i4 - i2) / (i1 - i3));
            }
            else if (i1 > i3 && i4 < i2) { // 第四象限
                phase.at<float>(g, k) = std::atan((i4 - i2) / (i1 - i3)) + 2 * CV_PI;
            }
            else {
                // 默认情况
                phase.at<float>(g, k) = 0;
            }
        }
    }

    return phase;
}

cv::Mat ThreeFrequencyUnwrapping::unwrapPhaseByThreeFrequencyFourPhase()
{
    // 存储3组共计12张图
    std::vector<std::vector<cv::Mat>> C(3, std::vector<cv::Mat>(4));

    std::cout << "Generating sine patterns..." << std::endl;

    // 存储12张正弦图案
    for (int i = 0; i < 3; i++) 
    { // 三种频率
        for (int j = 0; j < 4; j++) 
        { // 四种相位
            C[i][j] = generateSinePattern(i, j);
            //实际情况需是采集的图像
        }
    }

    // 对灰度值进行归一化处理
    for (int i = 0; i < 3; i++) 
    {
        for (int j = 0; j < 4; j++) 
        {
            C[i][j] = normalizeImage(C[i][j]);
        }
    }

    std::cout << "Computing wrapped phases..." << std::endl;

    // 计算三种频率的包裹相位
    std::vector<cv::Mat> phi(3);
    for (int i = 0; i < 3; i++) {
        phi[i] = computeWrappedPhase(C[i][0], C[i][1], C[i][2], C[i][3]);
    }

    cv::Mat PH1 = phi[0]; // 频率1的相位主值
    cv::Mat PH2 = phi[1]; // 频率2的相位主值  
    cv::Mat PH3 = phi[2]; // 频率3的相位主值

    std::cout << "Computing phase differences..." << std::endl;

    // 计算相位差
    cv::Mat PH12 = cv::Mat::zeros(height_, width_, CV_32F);
    cv::Mat PH23 = cv::Mat::zeros(height_, width_, CV_32F);
    cv::Mat PH123 = cv::Mat::zeros(height_, width_, CV_32F);

    for (int g = 0; g < height_; g++) 
    {
        for (int k = 0; k < width_; k++) 
        {
            PH12.at<float>(g, k) = safePhaseDifference(PH1.at<float>(g, k),PH2.at<float>(g, k));
            PH23.at<float>(g, k) = safePhaseDifference(PH2.at<float>(g, k),PH3.at<float>(g, k));
        }
    }

    std::cout << "Performing phase unwrapping..." << std::endl;

    // 计算最终相差（相位解包裹）
    for (int g = 0; g < height_; g++) 
    {
        for (int k = 0; k < width_; k++) 
        {
            PH123.at<float>(g, k) = safePhaseDifference(PH12.at<float>(g, k), PH23.at<float>(g, k));
        }
    }

    // 显示包裹及展开结果
    //displayResults(PH1, PH2, PH3, PH12, PH23, PH123);

    // 保存结果
    saveResults(PH1, PH2, PH3, PH12, PH23, PH123);

    return PH123;
}

// 显示结果图像
void ThreeFrequencyUnwrapping::displayResults(const cv::Mat& PH1, const cv::Mat& PH2, const cv::Mat& PH3,
    const cv::Mat& PH12, const cv::Mat& PH23, const cv::Mat& PH123)
{
    // 转换为显示格式 [0, 1] 范围
    cv::Mat display_PH1, display_PH2, display_PH3;
    cv::Mat display_PH12, display_PH23, display_PH123;

    cv::normalize(PH1, display_PH1, 0, 1, cv::NORM_MINMAX);
    cv::normalize(PH2, display_PH2, 0, 1, cv::NORM_MINMAX);
    cv::normalize(PH3, display_PH3, 0, 1, cv::NORM_MINMAX);
    cv::normalize(PH12, display_PH12, 0, 1, cv::NORM_MINMAX);
    cv::normalize(PH23, display_PH23, 0, 1, cv::NORM_MINMAX);
    cv::normalize(PH123, display_PH123, 0, 1, cv::NORM_MINMAX);

    cv::imshow("1 Phase Principal Value", display_PH1);
    cv::imshow("2 Phase Principal Value", display_PH2);
    cv::imshow("3 Phase Principal Value", display_PH3);
    cv::imshow("1,2 Heterodyne", display_PH12);
    cv::imshow("2,3 Heterodyne", display_PH23);
    cv::imshow("1,2,3 Heterodyne", display_PH123);

    std::cout << "Press any key to close windows..." << std::endl;
    cv::waitKey(0);
}

void ThreeFrequencyUnwrapping::saveResults(const cv::Mat& PH1, const cv::Mat& PH2, const cv::Mat& PH3,
    const cv::Mat& PH12, const cv::Mat& PH23, const cv::Mat& PH123) 
{

    cv::Mat save_PH1, save_PH2, save_PH3;
    cv::Mat save_PH12, save_PH23, save_PH123;

    // 归一化到 [0, 255] 范围用于保存
    cv::normalize(PH1, save_PH1, 0, 255, cv::NORM_MINMAX, CV_8U);
    cv::normalize(PH2, save_PH2, 0, 255, cv::NORM_MINMAX, CV_8U);
    cv::normalize(PH3, save_PH3, 0, 255, cv::NORM_MINMAX, CV_8U);
    cv::normalize(PH12, save_PH12, 0, 255, cv::NORM_MINMAX, CV_8U);
    cv::normalize(PH23, save_PH23, 0, 255, cv::NORM_MINMAX, CV_8U);
    cv::normalize(PH123, save_PH123, 0, 255, cv::NORM_MINMAX, CV_8U);

    cv::imwrite("1_phase_principal_value.bmp", save_PH1);
    cv::imwrite("2_phase_principal_value.bmp", save_PH2);
    cv::imwrite("3_phase_principal_value.bmp", save_PH3);
    cv::imwrite("12_heterodyne.bmp", save_PH12);
    cv::imwrite("23_heterodyne.bmp", save_PH23);
    cv::imwrite("123_heterodyne.bmp", save_PH123);

    std::cout << "Results saved to disk." << std::endl;
}


int ThreeFrequencyUnwrapping::GetFileNum(const string& path, const string& extParameter)
{
    fs::path dir_path(path);
    if (!fs::exists(dir_path) || !fs::is_directory(dir_path))
    {
        std::cerr << "错误：路径不存在或不是一个目录：" << path << std::endl;
        return -1;
    }
    int bmp_count = 0;
    // 遍历目录中的所有文件
    for (const auto& entry : fs::directory_iterator(dir_path))
    {
        if (entry.is_regular_file())
        {
            std::string ext = entry.path().extension().string(); // 如 ".BMP" 或 ".bmp"
            if (ext == extParameter)
            {
                ++bmp_count;
            }
        }
    }
    return bmp_count;
}
