void visualizeCameraPosesWithOpen3D(const CaliResult& params12, const CaliResult& params23)
{

    visualization::Visualizer visualizer;
    visualizer.CreateVisualizerWindow("Camera Poses Visualization - Open3D", 1024, 768);

    // 计算从相机1到相机3的变换矩阵
    Eigen::Matrix4d eigen_mat12, eigen_mat23;
    for (int i = 0; i < 4; i++)
    {
        for (int j = 0; j < 4; j++)
        {
            eigen_mat12(i, j) = params12.transformMatrix.at<double>(i, j);
            eigen_mat23(i, j) = params23.transformMatrix.at<double>(i, j);
        }
    }
    Eigen::Matrix4d transform_1_to_3 = eigen_mat12 * eigen_mat23;


    // 为每个相机创建坐标系和视锥体
    vector<Eigen::Matrix4d> camera_poses = {
        Eigen::Matrix4d::Identity(),  // 相机1：单位矩阵，作为世界坐标系原点
        eigen_mat12,                  // 相机2：相对于相机1的位姿
        transform_1_to_3              // 相机3：相对于相机1的位姿
    };

    vector<Eigen::Vector3d> colors = {
        Eigen::Vector3d(1.0, 0.0, 0.0),  // 红色 - 相机1
        Eigen::Vector3d(0.0, 1.0, 0.0),  // 绿色 - 相机2  
        Eigen::Vector3d(0.0, 0.0, 1.0)   // 蓝色 - 相机3
    };

    // 添加每个相机的可视化
    for (size_t i = 0; i < camera_poses.size(); i++) {

        // 创建坐标系
        auto coordinate_frame = geometry::TriangleMesh::CreateCoordinateFrame(0.1);
        coordinate_frame->Transform(camera_poses[i]);

        // 添加到可视化器
        visualizer.AddGeometry(coordinate_frame);

    }

    // 添加相机之间的连接线
    vector<Eigen::Vector3d> line_points = {
        Eigen::Vector3d(0, 0, 0),  // 相机1位置
        camera_poses[1].block<3,1>(0,3),  // 相机2位置
        camera_poses[2].block<3,1>(0,3)   // 相机3位置
    };

    vector<Eigen::Vector2i> lines = {
        Eigen::Vector2i(0, 1),  // 相机1到相机2
        Eigen::Vector2i(1, 2)   // 相机2到相机3
    };

    auto line_set = make_shared<geometry::LineSet>();
    line_set->points_ = line_points;
    line_set->lines_ = lines;
    line_set->colors_ = { Eigen::Vector3d(1.0, 0.0, 0.0),  // 红色线
                         Eigen::Vector3d(0.0, 1.0, 0.0) }; // 绿色线

    visualizer.AddGeometry(line_set);

    // 设置背景色和视角
    visualizer.GetRenderOption().background_color_ = Eigen::Vector3d(1.0, 1.0, 1.0);
    visualizer.GetRenderOption().point_size_ = 3.0;

    // 运行可视化
    visualizer.Run();
    visualizer.DestroyVisualizerWindow();
}
