MultiCamJointOpt::CamParameters MultiCamJointOpt::MultiCamJointOptimization::Run(CaliResult& Cam1To2, CaliResult& Cam2To3, CaliResult& Cam3To4)
{
	//获取相机1坐标系的3D点
	CamParameters ret;
	vector<vector<Point3f>> Cam1_ObjPoints = Cam1To2.Cam1_RegisObjPoints;
	//获取相机1到相机i的的变换矩阵
	cv::Mat _Cam1To2 = Cam1To2.transformMatrix;
	cv::Mat _Cam1To3 = Cam1To2.transformMatrix * Cam2To3.transformMatrix;
	cv::Mat _Cam1To4 = Cam1To2.transformMatrix * Cam2To3.transformMatrix * Cam3To4.transformMatrix;



	return ret;

}
