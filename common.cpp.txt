// 获取目录中指定扩展名的文件数量
int Utils::GetFileNum(const string& path, const string& extParameter)
{
    fs::path dir_path(path);
    if (!fs::exists(dir_path) || !fs::is_directory(dir_path))
    {
        std::cerr << "错误：路径不存在或不是一个目录：" << path << std::endl;
        return -1;
    }
    int count = 0;
    for (const auto& entry : fs::directory_iterator(dir_path))
    {
        if (entry.is_regular_file())
        {
            std::string ext = entry.path().extension().string();
            if (ext == extParameter)
            {
                ++count;
            }
        }
    }
    return count;
}

void Utils::saveCalibrationResults(const string& filename, const CaliResult& params)
{
    FileStorage fs(filename, FileStorage::WRITE);

    fs << "cameraMatrix1" << params.cameraMatrix1;
    fs << "distCoeffs1" << params.distCoeffs1;
    fs << "cameraMatrix2" << params.cameraMatrix2;
    fs << "distCoeffs2" << params.distCoeffs2;
    fs << "R" << params.R;
    fs << "T" << params.T;
    fs << "TransformMatrix12" << params.transformMatrix;
    fs << "E" << params.E;
    fs << "F" << params.F;
    fs << "imageSize" << params.imageSize;
    fs << "ReprojectionError" << params.rms;

    fs.release();
    cout << "标定结果已保存到: " << filename << endl;
}

// 计算4*4变换矩阵的逆
Mat Utils::invertTransform(const Mat& transform)
{
    Mat inv = Mat::eye(4, 4, CV_64F);

    Mat R = transform(Rect(0, 0, 3, 3));
    Mat T = transform(Rect(3, 0, 1, 3));

    Mat R_inv = R.t();
    Mat T_inv = -R_inv * T;

    R_inv.copyTo(inv(Rect(0, 0, 3, 3)));
    T_inv.copyTo(inv(Rect(3, 0, 1, 3)));

    return inv;
}
double Utils::computeSingleReprojectionError(const vector<vector<Point3f>>& objectPoints,
    const vector<vector<Point2f>>& imagePoints,
    const Mat& cameraMatrix,
    const Mat& distCoeffs,
    const Mat& R, const Mat& T)
{
    double totalError = 0.0;
    int totalPoints = 0;

    for (size_t i = 0; i < objectPoints.size(); i++)
    {
        vector<Point2f> projectedPoints;
        projectPoints(objectPoints[i], R, T, cameraMatrix, distCoeffs, projectedPoints);

        for (size_t j = 0; j < imagePoints[i].size(); j++)
        {
            double error = norm(imagePoints[i][j] - projectedPoints[j]);
            totalError += error * error;
        }
        totalPoints += imagePoints[i].size();
    }

    return sqrt(totalError / totalPoints);
}

double Utils::computeStereoReprojectionError(const vector<vector<Point3f>>& objectPoints,
    const vector<vector<Point2f>>& imagePoints1,
    const vector<vector<Point2f>>& imagePoints2,
    const Mat& cameraMatrix1, const Mat& distCoeffs1,
    const Mat& cameraMatrix2, const Mat& distCoeffs2,
    const Mat& R, const Mat& T)
{
    Mat R1 = Mat::eye(3, 3, CV_64F);
    Mat T1 = Mat::zeros(3, 1, CV_64F);

    double error1 = computeSingleReprojectionError(objectPoints, imagePoints1,
        cameraMatrix1, distCoeffs1, R1, T1);
    double error2 = computeSingleReprojectionError(objectPoints, imagePoints2,
        cameraMatrix2, distCoeffs2, R, T);

    double totalError = 0.0;
    int totalPoints = 0;

    for (size_t i = 0; i < objectPoints.size(); i++)
    {
        vector<Point2f> projectedPoints1, projectedPoints2;

        projectPoints(objectPoints[i], R1, T1, cameraMatrix1, distCoeffs1, projectedPoints1);
        projectPoints(objectPoints[i], R, T, cameraMatrix2, distCoeffs2, projectedPoints2);

        for (size_t j = 0; j < objectPoints[i].size(); j++)
        {
            double e1 = norm(imagePoints1[i][j] - projectedPoints1[j]);
            double e2 = norm(imagePoints2[i][j] - projectedPoints2[j]);
            totalError += (e1 * e1 + e2 * e2);
        }
        totalPoints += objectPoints[i].size() * 2;
    }

    double overallRMSE = sqrt(totalError / totalPoints);

    cout << "=== 重投影误差分析 ===" << endl;
    cout << "左相机RMS误差: " << error1 << " 像素" << endl;
    cout << "右相机RMS误差: " << error2 << " 像素" << endl;
    cout << "总体RMS误差: " << overallRMSE << " 像素" << endl;
    cout << "=========================" << endl;

    return overallRMSE;
}
vector<vector<Point3f>> Utils::convertToCamera1Coordinate(double& overall_rms,
    const vector<vector<Point3f>>& objPoints,
    const vector<vector<Point2f>>& imagePoints1,
    const Mat& cameraMatrix1,
    const Mat& distCoeffs1)
{
    vector<vector<Point3f>> objPointsInCam1;

    double total_reproj_error = 0.0;
    int total_points = 0;
    vector<double> per_image_errors;

    cout << "=== PnP坐标转换及误差分析 ===" << endl;

    for (size_t i = 0; i < objPoints.size(); ++i)
    {
        // 使用PnP计算标定板到相机1的变换
        Mat rvec, tvec;
        bool success = solvePnP(objPoints[i], imagePoints1[i],
            cameraMatrix1, distCoeffs1, rvec, tvec);

        if (!success)
        {
            cerr << "PnP failed for image pair " << i << endl;
            continue;
        }

        Mat R_board_to_cam1;
        Rodrigues(rvec, R_board_to_cam1);

        // 将标定板点转换到相机1坐标系
        vector<Point3f> pointsInCam1;

        // 计算该图像的重投影误差
        vector<Point2f> projected_points;
        projectPoints(objPoints[i], rvec, tvec, cameraMatrix1, distCoeffs1, projected_points);

        double image_error = 0.0;

        for (size_t j = 0; j < objPoints[i].size(); ++j)
        {
            // 坐标转换
            Mat pt_board = (Mat_<double>(3, 1) << objPoints[i][j].x,
                objPoints[i][j].y,
                objPoints[i][j].z);
            Mat pt_cam1 = R_board_to_cam1 * pt_board + tvec;
            pointsInCam1.push_back(Point3f(pt_cam1.at<double>(0),
                pt_cam1.at<double>(1),
                pt_cam1.at<double>(2)));

            // 计算重投影误差
            double error = norm(imagePoints1[i][j] - projected_points[j]);
            image_error += error;
            total_reproj_error += error;
        }

        total_points += objPoints[i].size();
        double rms_error = image_error / objPoints[i].size();
        per_image_errors.push_back(rms_error);

        objPointsInCam1.push_back(pointsInCam1);

        cout << "图像 " << i << ": " << objPoints[i].size() << " 个点, "
            << "RMS平均误差: " << rms_error << " px" << endl;
    }

    if (total_points > 0)
    {
        overall_rms = total_reproj_error / total_points;
        double max_error = *max_element(per_image_errors.begin(), per_image_errors.end());
        double min_error = *min_element(per_image_errors.begin(), per_image_errors.end());
        double avg_error = accumulate(per_image_errors.begin(), per_image_errors.end(), 0.0)
            / per_image_errors.size();

        cout << "\n=== 总体统计 ===" << endl;
        cout << "总点数: " << total_points << endl;
        cout << "总图像数: " << objPointsInCam1.size() << endl;
        cout << "标定板->相机1 平均重投影误差: " << overall_rms << " px, "
            << "最小: " << min_error << " px, 最大: " << max_error << " px" << endl;
    }

    return objPointsInCam1;
}
