using namespace cv;
using namespace std;


struct ThreeFrequencyUnwrapping
{
    ThreeFrequencyUnwrapping(int width, int height, std::vector<int> freq, std::string imgpath);
    // index=false 采用生成的正弦图案
    // index=true  采用采集图像
    cv::Mat generateSinePattern(int freq_index, int phase_shift, bool index = false);
    // 对图像进行归一化处理 [0, 1] 范围
    cv::Mat normalizeImage(const cv::Mat& img);
    // 计算包裹相位（[0,2PI], 手动四象限反正切实现）
    cv::Mat computeWrappedPhase(const cv::Mat& I1, const cv::Mat& I2, const cv::Mat& I3, const cv::Mat& I4);
    // 相位差计算
    const float safePhaseDifference(float phi1, float phi2) 
    {
        if (phi1 >= phi2) {
            return phi1 - phi2;
        }
        else {
            return phi1 + 2 * CV_PI - phi2;
        }
    }
    // 接口：执行四步获取包裹图、三频外差对相位展开
    // 注意：水平相位和垂直相位分开展开，需调用两次
    cv::Mat unwrapPhaseByThreeFrequencyFourPhase();

    // 显示结果图像
    void displayResults(const cv::Mat& PH1, const cv::Mat& PH2, const cv::Mat& PH3,
        const cv::Mat& PH12, const cv::Mat& PH23, const cv::Mat& PH123);
   
    // 保存包裹及相位展开结果图像
    void saveResults(const cv::Mat& PH1, const cv::Mat& PH2, const cv::Mat& PH3,
        const cv::Mat& PH12, const cv::Mat& PH23, const cv::Mat& PH123);

    int GetFileNum(const string& path, const string& extParameter);

    int width_;
    int height_;
    std::vector<int> freq_;//整个宽度内的条纹数
    std::string imgpath_;
    //CalibrationByMirror* calibByMir;

};
