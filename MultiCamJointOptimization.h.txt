    // 相机位姿优化的残差函数
    struct ReprojectionError
    {
        ReprojectionError(const Point3f& objectPoint, const Point2f& imagePoint,
            double fx, double fy, double cx, double cy,
            double k1, double k2, double p1, double p2, double k3)
            : objectPoint(objectPoint), imagePoint(imagePoint),
            fx(fx), fy(fy), cx(cx), cy(cy), k1(k1), k2(k2), p1(p1), p2(p2), k3(k3) {
        }

        template <typename T>
        bool operator()(const T* const R, const T* const t, T* residuals) const
        {
            // 将3D点从世界坐标系变换到相机坐标系
            T point[3] = { T(objectPoint.x), T(objectPoint.y), T(objectPoint.z) };
            T camera_point[3] = { T(0), T(0), T(0) };

            // 应用旋转和平移：P_camera = R * P_world + t
            for (int i = 0; i < 3; i++) {
                for (int j = 0; j < 3; j++) {
                    camera_point[i] += R[i * 3 + j] * point[j];
                }
                camera_point[i] += t[i];
            }

            T x = camera_point[0];
            T y = camera_point[1];
            T z = camera_point[2];

            // 检查深度是否为正
            if (z <= T(1e-6))
            {
                residuals[0] = T(1000);  // 大的残差惩罚
                residuals[1] = T(1000);
                return true;
            }

            // 归一化坐标
            T x_normalized = x / z;
            T y_normalized = y / z;

            // 径向畸变
            T r2 = x_normalized * x_normalized + y_normalized * y_normalized;
            T radial_distortion = T(1) + T(k1) * r2 + T(k2) * r2 * r2 + T(k3) * r2 * r2 * r2;

            // 切向畸变
            T tangential_x = T(2) * T(p1) * x_normalized * y_normalized + T(p2) * (r2 + T(2) * x_normalized * x_normalized);
            T tangential_y = T(p1) * (r2 + T(2) * y_normalized * y_normalized) + T(2) * T(p2) * x_normalized * y_normalized;

            // 畸变后的坐标
            T x_distorted = x_normalized * radial_distortion + tangential_x;
            T y_distorted = y_normalized * radial_distortion + tangential_y;

            // 投影到像素坐标
            T predicted_x = T(fx) * x_distorted + T(cx);
            T predicted_y = T(fy) * y_distorted + T(cy);

            // 计算残差
            residuals[0] = predicted_x - T(imagePoint.x);
            residuals[1] = predicted_y - T(imagePoint.y);

            return true;
        }

        static ceres::CostFunction* Create(const Point3f& objectPoint, const Point2f& imagePoint,
            const Mat& cameraMatrix, const Mat& distCoeffs)
        {
            // 提取内参和畸变参数
            double fx = cameraMatrix.at<double>(0, 0);
            double fy = cameraMatrix.at<double>(1, 1);
            double cx = cameraMatrix.at<double>(0, 2);
            double cy = cameraMatrix.at<double>(1, 2);

            double k1 = 0.0, k2 = 0.0, p1 = 0.0, p2 = 0.0, k3 = 0.0;
            if (distCoeffs.rows > 0 && distCoeffs.cols > 0)
            {
                k1 = distCoeffs.rows >= 1 ? distCoeffs.at<double>(0, 0) : 0.0;
                k2 = distCoeffs.rows >= 2 ? distCoeffs.at<double>(1, 0) : 0.0;
                p1 = distCoeffs.rows >= 3 ? distCoeffs.at<double>(2, 0) : 0.0;
                p2 = distCoeffs.rows >= 4 ? distCoeffs.at<double>(3, 0) : 0.0;
                k3 = distCoeffs.rows >= 5 ? distCoeffs.at<double>(4, 0) : 0.0;
            }

            return new ceres::AutoDiffCostFunction<ReprojectionError, 2, 9, 3>(
                new ReprojectionError(objectPoint, imagePoint, fx, fy, cx, cy, k1, k2, p1, p2, k3));
        }

    private:
        Point3f objectPoint;
        Point2f imagePoint;
        double fx, fy, cx, cy;  // 内参
        double k1, k2, p1, p2, k3;  // 畸变系数
    };
