CharArUco::CharArUco(string& ImgPathLeft, string& ImgPathRight, string& outputFilePath, int& boardWidth, int& boardHeight, float& CheckerSize, float& MarkerSize, double& focalLength, double& pixelSize)
{
    int imgnumLeft = Utils::GetFileNum(ImgPathLeft, ".bmp");
    if (imgnumLeft != Utils::GetFileNum(ImgPathRight, ".bmp"))
    {
        cout << "左右相机图片数量不一致" << endl;
        return;
    }

    ImgNum = imgnumLeft;
    m_ImgPathLeft = ImgPathLeft;
    m_ImgPathRight = ImgPathRight;
    m_boardWidth = boardWidth;
    m_boardHeight = boardHeight;
    m_CheckerSize = CheckerSize;
    m_MarkerSize = MarkerSize;

    m_focalLength = focalLength;
    m_pixelSize = pixelSize;

    double fx_pixels = m_focalLength / m_pixelSize;
    double fy_pixels = m_focalLength / m_pixelSize;

    //// 设置内参矩阵初值,用于精细化ChArUco角点Id
    m_CaliResult.cameraMatrix1.at<double>(0, 0) = fx_pixels;  // fx
    m_CaliResult.cameraMatrix1.at<double>(1, 1) = fy_pixels;  // fy
    m_CaliResult.cameraMatrix1.at<double>(0, 2) = m_CaliResult.imageSize.width / 2.0;
    m_CaliResult.cameraMatrix1.at<double>(1, 2) = m_CaliResult.imageSize.height / 2.0;

    m_CaliResult.cameraMatrix2.at<double>(0, 0) = fx_pixels;  // fx
    m_CaliResult.cameraMatrix2.at<double>(1, 1) = fy_pixels;  // fy
    m_CaliResult.cameraMatrix2.at<double>(0, 2) = m_CaliResult.imageSize.width / 2.0;
    m_CaliResult.cameraMatrix2.at<double>(1, 2) = m_CaliResult.imageSize.height / 2.0;
    
}

bool CharArUco::detectCharucoCorners(Mat& image, Ptr<aruco::CharucoBoard>& board,
    vector<int>& charucoIds, vector<Point2f>& charucoCorners,
    const Mat& cameraMatrix, const Mat& distCoeffs)
{
    vector<int> markerIds;
    vector<vector<Point2f>> markerCorners, rejected;
    Ptr<aruco::DetectorParameters> detectorParams = aruco::DetectorParameters::create();
    detectorParams->cornerRefinementMethod = aruco::CORNER_REFINE_SUBPIX;

    Mat gray;
    if (image.channels() == 3) 
        cvtColor(image, gray, COLOR_BGR2GRAY);
    else gray = image;

    // 检测 ArUco 标记
    aruco::detectMarkers(gray, board->dictionary, markerCorners, markerIds, detectorParams, rejected);

    if (markerIds.empty())
        return false;

    // 精细化、恢复标记
    aruco::refineDetectedMarkers(gray, board, markerCorners, markerIds, rejected, cameraMatrix, distCoeffs);

    // 插值 ChArUco 角点
    aruco::interpolateCornersCharuco(markerCorners, markerIds, gray, board,
        charucoCorners, charucoIds, cameraMatrix, distCoeffs);

    return charucoIds.size() >= 4;
}

//执行双目标定
CaliResult CharArUco::stereoCalibration(Ptr<aruco::Dictionary>& dictionary, Ptr<aruco::CharucoBoard>& board)
{
    //存储配准好的检测结果
    vector<vector<Point2f>> RegisCharucoCorners1, RegisCharucoCorners2;
    vector<vector<Point3f>> RegisObjPoints;
    //检测所有采集的标定图像
    bool IsAllCorners = detectAllCharucoCorners(RegisObjPoints, RegisCharucoCorners1, RegisCharucoCorners2, dictionary);

    m_CaliResult.RegisCharucoCorners1 = RegisCharucoCorners1;
    m_CaliResult.RegisCharucoCorners2 = RegisCharucoCorners2;

    if (!IsAllCorners)
    {
        cout << "Failed is to detectAllCharucoCorners: False, return! " << endl;
        return m_CaliResult;
    }

    // ========== 单目标定 ==========
    calculateIntriinsic();
    // ========== 执行双目标定 ==========
    double rms = stereoCalibrate(
        RegisObjPoints, RegisCharucoCorners1, RegisCharucoCorners2,
        m_CaliResult.cameraMatrix1, m_CaliResult.distCoeffs1,
        m_CaliResult.cameraMatrix2, m_CaliResult.distCoeffs2,
        m_CaliResult.imageSize,
        m_CaliResult.R, m_CaliResult.T, m_CaliResult.E, m_CaliResult.F,
        CALIB_FIX_INTRINSIC,  // 固定内参和畸变系数
        //CALIB_USE_INTRINSIC_GUESS//使用初始值并继续优化内参
        TermCriteria(TermCriteria::COUNT + TermCriteria::EPS, 100, 1e-10)
    );

    std::cout << "Stereo calibration RMS error: " << rms << std::endl;
    m_CaliResult.rms = rms;

    std::cout << "Start Bundle Adjustment optimization..." << std::endl;
    RunBundleAdjustment(RegisObjPoints, RegisCharucoCorners1, RegisCharucoCorners2);

    //外参结果为：从cam1变换到cam2，故此处取逆即从cam2变换到cam1
     // 将R和T合并为4x4变换矩阵
    Mat transformMatrix = Mat::eye(4, 4, CV_64F);
    m_CaliResult.R.copyTo(transformMatrix(Rect(0, 0, 3, 3)));
    m_CaliResult.T.copyTo(transformMatrix(Rect(3, 0, 1, 3)));
    m_CaliResult.transformMatrix = Utils::invertTransform(transformMatrix);
    //m_CaliResult.transformMatrix = (transformMatrix);

    return m_CaliResult;
}
// 计算单相机的重投影误差
double computeReprojectionError(const vector<vector<Point3f>>& objectPoints,const vector<vector<Point2f>>& imagePoints,const Mat& cameraMatrix,const Mat& distCoeffs,const Mat& R, const Mat& T) 
{
    double totalError = 0.0;
    int totalPoints = 0;

    for (int i = 0; i < objectPoints.size(); i++) 
    {
        vector<Point2f> projectedPoints;

        // 将3D点投影到图像平面
        projectPoints(objectPoints[i], R, T, cameraMatrix, distCoeffs, projectedPoints);

        // 计算该图像中所有点的误差
        for (int j = 0; j < imagePoints[i].size(); j++) 
           {
            double error = norm(imagePoints[i][j] - projectedPoints[j]);
            totalError += error * error;  // 平方误差
        }
        totalPoints += imagePoints[i].size();
    }

    return sqrt(totalError / totalPoints);  // RMS误差
}

// 计算立体相机的重投影误差
double  CharArUco::computeStereoReprojectionError(const vector<vector<Point3f>>& objectPoints,
    const vector<vector<Point2f>>& imagePoints1,
    const vector<vector<Point2f>>& imagePoints2,
    const Mat& cameraMatrix1, const Mat& distCoeffs1,
    const Mat& cameraMatrix2, const Mat& distCoeffs2,
    const Mat& R, const Mat& T) {

    // 计算左相机误差
    Mat R1 = Mat::eye(3, 3, CV_64F);
    Mat T1 = Mat::zeros(3, 1, CV_64F);
    double error1 = computeReprojectionError(objectPoints, imagePoints1,cameraMatrix1, distCoeffs1, R1, T1);

    // 计算右相机误差
    double error2 = computeReprojectionError(objectPoints, imagePoints2,
        cameraMatrix2, distCoeffs2, R, T);

    // 计算总体误差
    double totalError = 0.0;
    int totalPoints = 0;

    for (size_t i = 0; i < objectPoints.size(); i++) 
    {
        vector<Point2f> projectedPoints1, projectedPoints2;

        projectPoints(objectPoints[i], R1, T1, cameraMatrix1, distCoeffs1, projectedPoints1);
        projectPoints(objectPoints[i], R, T, cameraMatrix2, distCoeffs2, projectedPoints2);

        for (size_t j = 0; j < objectPoints[i].size(); j++) 
        {
            double error1 = norm(imagePoints1[i][j] - projectedPoints1[j]);
            double error2 = norm(imagePoints2[i][j] - projectedPoints2[j]);
            totalError += (error1 * error1 + error2 * error2);
        }
        totalPoints += objectPoints[i].size() * 2;
    }

    double overallRMSE = sqrt(totalError / totalPoints);

    cout << "=== 重投影误差分析 ===" << endl;
    cout << "左相机RMS误差: " << error1 << " 像素" << endl;
    cout << "右相机RMS误差: " << error2 << " 像素" << endl;
    cout << "总体RMS误差: " << overallRMSE << " 像素" << endl;
    cout << "=========================" << endl;
    return overallRMSE;
}

vector<vector<Point3f>>CharArUco:: convertToCamera1Coordinate(double& overall_rms, const vector<vector<Point3f>>& RegisObjPoints, const vector<vector<Point2f>>& RegisCharucoCorners1, const Mat& cameraMatrix1, const Mat& distCoeffs1)
{
    vector<vector<Point3f>> objPointsInCam1;

    double total_reproj_error = 0.0;
    int total_points = 0;
    vector<double> per_image_errors;

    cout << "=== PnP坐标转换及误差分析 ===" << endl;

    for (size_t i = 0; i < RegisObjPoints.size(); ++i) {
        // 使用PnP计算标定板到相机1的变换
        Mat rvec, tvec;
        bool success = solvePnP(RegisObjPoints[i], RegisCharucoCorners1[i],cameraMatrix1, distCoeffs1, rvec, tvec);

        if (!success) {
            cerr << "PnP failed for image pair " << i << endl;
            continue;
        }

        Mat R_board_to_cam1;
        Rodrigues(rvec, R_board_to_cam1);

        // 将标定板点转换到相机1坐标系
        vector<Point3f> pointsInCam1;

        // 计算该图像的重投影误差
        vector<Point2f> projected_points;
        projectPoints(RegisObjPoints[i], rvec, tvec, cameraMatrix1, distCoeffs1, projected_points);

        double image_error = 0.0;

        for (size_t j = 0; j < RegisObjPoints[i].size(); ++j) 
        {
            // 坐标转换
            Mat pt_board = (Mat_<double>(3, 1) << RegisObjPoints[i][j].x, RegisObjPoints[i][j].y, RegisObjPoints[i][j].z);
            Mat pt_cam1 = R_board_to_cam1 * pt_board + tvec;
            pointsInCam1.push_back(Point3f(pt_cam1.at<double>(0), pt_cam1.at<double>(1), pt_cam1.at<double>(2)));

            // 计算重投影误差
            double error = norm(RegisCharucoCorners1[i][j] - projected_points[j]);
            image_error += error;
            total_reproj_error += error;
        }

        total_points += RegisObjPoints[i].size();
        double rms_error = (image_error / RegisObjPoints[i].size());
        per_image_errors.push_back(rms_error);

        objPointsInCam1.push_back(pointsInCam1);

        // 输出该图像的误差信息
        cout << "图像 " << i << ": " << RegisObjPoints[i].size() << " 个点, "<< "RMS平均误差: " << rms_error << " px, " << endl;
    }

    // 输出总体统计信息
    if (total_points > 0) 
    {
        overall_rms = total_reproj_error / total_points;
        double max_image_error = *max_element(per_image_errors.begin(), per_image_errors.end());
        double min_image_error = *min_element(per_image_errors.begin(), per_image_errors.end());
        double avg_image_error = accumulate(per_image_errors.begin(), per_image_errors.end(), 0.0) / per_image_errors.size();

        cout << "\n=== 总体统计 ===" << endl;
        cout << "总点数: " << total_points << endl;
        cout << "总图像数: " << objPointsInCam1.size() << endl;
        cout << "标定板->相机1, 平均重投影误差: " << overall_rms << " px" << 
            " 最小: " << min_image_error << " px" << "最大: " << max_image_error << " px" << endl;
        /******************************************************/
        //可增加过滤功能，将误差较大的图像排除，再进行BA优化
        //去除图像时，需要左右图像同时去除，较麻烦，懒得搞。
    }

    return objPointsInCam1;
}

void CharArUco::RunBundleAdjustment(const vector<vector<Point3f>>& RegisObjPoints, const vector<vector<Point2f>>& RegisCharucoCorners1, const vector<vector<Point2f>>& RegisCharucoCorners2)
{
    // 使用Ceres进行Bundle Adjustment
    ceres::Problem problem;
    // 将标定板坐标系下的点转换到相机1坐标系
    vector<vector<Point3f>> Cam1_RegisObjPoints;
    double rms_pnp_cam1 = 0.0;//记录坐标系转换误差
    Cam1_RegisObjPoints = convertToCamera1Coordinate(rms_pnp_cam1,RegisObjPoints, RegisCharucoCorners1, m_CaliResult.cameraMatrix1, m_CaliResult.distCoeffs1);
    
    //设置世界坐标系到相机1坐标系的变换矩阵
    cv::Mat R1_mat = cv::Mat::eye(3, 3, CV_64F);
    cv::Mat T1_mat = cv::Mat::zeros(3, 1, CV_64F);
/*    double R1[9] = { 1, 0, 0, 0, 1, 0, 0, 0, 1 };  
    double T1[3] = { 0, 0, 0 };   */             

    //相机2的外参（从相机1到相机2的变换，待优化）
    Mat R2_mat = m_CaliResult.R.clone();
    Mat T2_mat = m_CaliResult.T.clone();


    // 为每个标定点创建一个残差块
    for (size_t i = 0; i < Cam1_RegisObjPoints.size(); ++i)
    {
        for (size_t j = 0; j < Cam1_RegisObjPoints[i].size(); ++j)
        {
            Point3f objectPoint = Cam1_RegisObjPoints[i][j];
            Point2f imagePoint1 = RegisCharucoCorners1[i][j];
            Point2f imagePoint2 = RegisCharucoCorners2[i][j];

            // 创建相机1的残差块
            ceres::CostFunction* costFunction1 = ReprojectionError::Create(objectPoint, imagePoint1, m_CaliResult.cameraMatrix1, m_CaliResult.distCoeffs1);
            // 创建相机2的残差块
            ceres::CostFunction* costFunction2 = ReprojectionError::Create(objectPoint, imagePoint2, m_CaliResult.cameraMatrix2, m_CaliResult.distCoeffs2);
            // 将残差块添加到问题中
            // 固定相机1的外参，优化相机2的外参
            problem.AddResidualBlock(costFunction1, nullptr, R1_mat.ptr<double>(), T1_mat.ptr<double>());
            problem.AddResidualBlock(costFunction2, nullptr, R2_mat.ptr<double>(), T2_mat.ptr<double>());
            //problem.AddResidualBlock(costFunction2, nullptr, R2, T2);
        }
    }
    // 经测试发现不固定相机1的外参，重投影误差会较大
    problem.SetParameterBlockConstant(R1_mat.ptr<double>());
    problem.SetParameterBlockConstant(T1_mat.ptr<double>());

    // 设置问题求解器选项
    ceres::Solver::Options options;
    options.linear_solver_type = ceres::DENSE_SCHUR;  // 使用稠密的Schur求解器
    options.minimizer_type = ceres::TRUST_REGION;    // 使用信赖域方法
    options.max_num_iterations = 100;  // 最大迭代次数

    options.function_tolerance = 1e-8;
    options.gradient_tolerance = 1e-10;
    options.parameter_tolerance = 1e-8;

    // 执行优化
    ceres::Solver::Summary summary;
    ceres::Solve(options, &problem, &summary);

    // 输出优化结果
    std::cout << summary.BriefReport() << std::endl;

    cout << "\n优化后重投影误差:" << endl;
    double BA_ReProErr=computeStereoReprojectionError(Cam1_RegisObjPoints, RegisCharucoCorners1, RegisCharucoCorners2,m_CaliResult.cameraMatrix1, m_CaliResult.distCoeffs1,m_CaliResult.cameraMatrix2, m_CaliResult.distCoeffs2,m_CaliResult.R, m_CaliResult.T);
    
    BA_ReProErr = BA_ReProErr - rms_pnp_cam1;
    cout << "BA_ReProErr = " << BA_ReProErr << endl;
    if (BA_ReProErr < m_CaliResult.rms)
    {
        m_CaliResult.rms = BA_ReProErr;
        m_CaliResult.T= T2_mat.clone();
        m_CaliResult.R = R2_mat.clone();
    }
}


unordered_map<int, Point2f>CharArUco::cornersMap(vector<int>ids2, vector<Point2f>corners2)
{
    unordered_map<int, Point2f>corners_idsMap;
    for (int i = 0; i < ids2.size(); i++)
    {
        corners_idsMap[ids2[i]] = corners2[i];
    }
    return corners_idsMap;
}
